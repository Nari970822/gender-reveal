  <script>
    function popBalloon() {
      // ★★★ ここを 'boy' または 'girl' に変更してください ★★★
      const gender = 'girl';

      const balloon = document.getElementById('balloon');
      const result = document.getElementById('result');
      
      balloon.classList.add('popped');
      
      setTimeout(() => {
        if (gender === 'boy') {
          result.textContent = "It's a Boy!";
          result.className = 'boy show';
        } else {
          result.textContent = "It's a Girl!";
          result.className = 'girl show';
        }
      }, 300); // アニメーションの後に表示
    }
  </script>
